 40    @staticmethod
 41    def _normalize_datetime(date_value, field_name):
 42        """Parse and normalize a date/datetime value to a naive UTC datetime.
 43
 44        Accepts datetime objects, date objects, and ISO 8601 strings.
 45        Timezone-aware values are converted to UTC and made naive.
 46
 47        Args:
 48            date_value: The value to normalize.
 49            field_name: Name of the field (for error messages).
 50
 51        Returns:
 52            A naive datetime in UTC, or None if date_value is None.
 53
 54        Raises:
 55            ValueError: If the value cannot be parsed.
 56        """
 57        if date_value is None:
 58            return None
 59
 60        if isinstance(date_value, datetime):
 61            value = date_value
 62        elif isinstance(date_value, date):
 63            value = datetime.combine(date_value, time.min)
 64        elif isinstance(date_value, str):
 65            normalized_value = date_value.strip().replace('Z', '+00:00')
 66            try:
 67                value = datetime.fromisoformat(normalized_value)
 68            except ValueError:
 69                try:
 70                    parsed_date = datetime.strptime(date_value, '%Y-%m-%d').date()
 71                except ValueError:
 72                    raise ValueError(f"Invalid {field_name} format. Use ISO 8601 datetime values")
 73                value = datetime.combine(parsed_date, time.min)
 74        else:
 75            raise ValueError(f"Invalid {field_name} format. Use ISO 8601 datetime values")
 76
 77        if value.tzinfo is not None:
 78            value = value.astimezone(timezone.utc).replace(tzinfo=None)
 79
 80        return value

 82    @staticmethod
 83    def _format_datetime(value):
 84        """Format a datetime as a human-readable UTC string.
 85
 86        Args:
 87            value: A datetime to format.
 88
 89        Returns:
 90            String like "Apr 21, 2026, 2:30 PM UTC".
 91        """
 92        if value.tzinfo is None:
 93            value = value.replace(tzinfo=timezone.utc)
 94        else:
 95            value = value.astimezone(timezone.utc)
 96        hour = value.hour % 12 or 12
 97        minute = f'{value.minute:02d}'
 98        period = 'AM' if value.hour < 12 else 'PM'
 99        return f"{value.strftime('%b')} {value.day}, {value.year}, {hour}:{minute} {period} UTC"

102    def _equipment_has_conflict(equipment_id, start_date, end_date):
103        """Check if an equipment item has a date conflict with existing rentals.
104
105        Looks for non-deleted rentals in blocking statuses (requesting or active)
106        that overlap the given date range.
107
108        Args:
109            equipment_id: The equipment's primary key.
110            start_date: Start of the proposed date range.
111            end_date: End of the proposed date range.
112
113        Returns:
114            The conflicting Rental instance, or None if no conflict.
115        """
116        conflict = (
117            db.session.query(Rental)
118            .join(RentalHasEquipment, Rental.id == RentalHasEquipment.rental_id)
119            .filter(
120                RentalHasEquipment.equipment_id == equipment_id,
121                Rental.deleted.is_(False),
122                Rental.status.in_(BLOCKING_STATUSES),
123                Rental.start_date <= end_date,
124                Rental.end_date >= start_date,
125            )
126            .first()
127        )
128        return conflict

130    @staticmethod
131    def create_rental(
132        renter_id,
133        vendor_id,
134        agreed_price,
135        start_date=None,
136        end_date=None,
137        location=None,
138        status='requesting',
139        deleted=False,
140        equipment_ids=None,
141        meeting_lat=None,
142        meeting_lng=None,
143    ):
144        """Create a new rental request with equipment.
145
146        Validates participants, dates, equipment ownership, and availability.
147        The renter is auto-approved; the vendor must separately approve.
148
149        Args:
150            renter_id: Primary key of the renter User.
151            vendor_id: Primary key of the vendor User.
152            agreed_price: Negotiated price in USD.
153            start_date: Rental start (ISO string or datetime, min 2 hours ahead).
154            end_date: Rental end (ISO string or datetime, must be after start).
155            location: Meeting location text.
156            status: Initial status (default 'requesting').
157            deleted: Soft-delete flag.
158            equipment_ids: List of Equipment primary keys to include.
159            meeting_lat: Latitude of meeting point.
160            meeting_lng: Longitude of meeting point.
161
162        Returns:
163            The created Rental instance.
164
165        Raises:
166            ValueError: If validation fails (missing fields, self-rental,
167                date conflicts, equipment not found or not owned by vendor).
168        """
169        if not all([renter_id, vendor_id, agreed_price]):
170            raise ValueError("renter_id, vendor_id, and agreed_price are required")
171        if renter_id == vendor_id:
172            raise ValueError("Renter and vendor cannot be the same user")
173
174        minimum_start_date = datetime.utcnow() + RentalService.MIN_REQUEST_LEAD_TIME
175
176        # If dates are omitted by the request flow, default to a 1-day request window.
177        if not start_date:
178            start_date = minimum_start_date
179        else:
180            start_date = RentalService._normalize_datetime(start_date, 'start_date')
181
182        if not end_date:
183            end_date = start_date + timedelta(days=1)
184        else:
185            end_date = RentalService._normalize_datetime(end_date, 'end_date')
186
187        if start_date < minimum_start_date:
188            raise ValueError("Start date must be at least 2 hours in the future")
189
190        if end_date <= start_date:
191            raise ValueError("End date must be after start date")
192        if status not in VALID_RENTAL_STATUSES:
193            raise ValueError("Invalid rental status")
194
195        requested_equipment_ids = []
196        if equipment_ids and isinstance(equipment_ids, list):
197            requested_equipment_ids.extend(equipment_ids)
198
199        # Normalize to integer IDs and flatten any accidental nested lists.
200        normalized_equipment_ids = []
201        for raw_id in requested_equipment_ids:
202            if isinstance(raw_id, list):
203                normalized_equipment_ids.extend(raw_id)
204            else:
205                normalized_equipment_ids.append(raw_id)
206
207        try:
208            normalized_equipment_ids = [int(eid) for eid in normalized_equipment_ids]
209        except (TypeError, ValueError):
210            raise ValueError("equipment_ids must be integers")
211
212        # Deduplicate while preserving order.
213        deduped_equipment_ids = list(dict.fromkeys(normalized_equipment_ids))
214        selected_equipment = []
215        for eid in deduped_equipment_ids:
216            equipment = Equipment.query.get(eid)
217            if not equipment:
218                raise ValueError(f"Equipment not found: {eid}")
219            if equipment.owner_id != vendor_id:
220                raise ValueError(f"Equipment does not belong to selected vendor: {eid}")
221            conflict = RentalService._equipment_has_conflict(equipment.id, start_date, end_date)
222            if conflict:
223                raise ValueError(f"Equipment is unavailable for selected dates: {equipment.name}")
224            selected_equipment.append(equipment)
225        
226        rental = Rental(
227            renter_id=renter_id,
228            vendor_id=vendor_id,
229            agreed_price=agreed_price,
230            start_date=start_date,
231            end_date=end_date,
232            location=location,
233            meeting_lat=meeting_lat,
234            meeting_lng=meeting_lng,
235            status=status,
236            renter_approved=True,
237            vendor_approved=False,
238            deleted=deleted,
239            renter_reviewed=False,
240            vendor_reviewed=False
241        )
242        db.session.add(rental)
243        db.session.flush()
244
245        for equipment in selected_equipment:
246            db.session.add(RentalHasEquipment(equipment_id=equipment.id, rental_id=rental.id, equipment_reviewed=False))
247
248        db.session.commit()
249        return rental

340    @staticmethod
341    def get_vendor_equipment_with_availability(vendor_id, start_date, end_date):
342        """Check availability of all vendor equipment for a date range.
343
344        Args:
345            vendor_id: The vendor User's primary key.
346            start_date: Start of the proposed date range (ISO string or datetime).
347            end_date: End of the proposed date range (ISO string or datetime).
348
349        Returns:
350            List of equipment dicts, each with ``available`` (bool) and
351            ``unavailable_reason`` (string or None).
352
353        Raises:
354            ValueError: If dates are missing or invalid.
355        """
356        start_date = RentalService._normalize_datetime(start_date, 'start_date')
357        end_date = RentalService._normalize_datetime(end_date, 'end_date')
358        if not start_date or not end_date:
359            raise ValueError("start_date and end_date are required")
360        if end_date <= start_date:
361            raise ValueError("End date must be after start date")
362
363        equipment_list = Equipment.query.filter_by(owner_id=vendor_id).all()
364        result = []
365        for equipment in equipment_list:
366            conflict = RentalService._equipment_has_conflict(equipment.id, start_date, end_date)
367            data = equipment.to_dict()
368            data['available'] = conflict is None
369            data['unavailable_reason'] = None if conflict is None else (
370                f"Booked {RentalService._format_datetime(conflict.start_date)} to {RentalService._format_datetime(conflict.end_date)} ({conflict.status})"
371            )
372            result.append(data)
373
374        return result

464    @staticmethod
465    def update_rental(
466        rental_id,
467        status=None,
468        location=None,
469        agreed_price=None,
470        deleted=None,
471        meeting_lat=None,
472        meeting_lng=None,
473        start_date=None,
474        end_date=None,
475        actor_user_id=None,
476        approve=False,
477    ):
478        """Update a rental's status, terms, and/or approval.
479
480        Handles the full spectrum of rental updates: status transitions,
481        term changes (price, dates, location), and approval management.
482        When terms change, both parties must re-approve. Enforces business
483        rules like cancellation permissions, return timing, and
484        renegotiation lead time.
485
486        Args:
487            rental_id: The rental's primary key.
488            status: New status value.
489            location: New meeting location text.
490            agreed_price: New agreed price.
491            deleted: New soft-delete flag.
492            meeting_lat: New meeting latitude.
493            meeting_lng: New meeting longitude.
494            start_date: New start date (ISO string or datetime).
495            end_date: New end date (ISO string or datetime).
496            actor_user_id: Primary key of the user performing the update.
497            approve: If True, record the actor's approval.
498
499        Returns:
500            The updated Rental instance.
501
502        Raises:
503            ValueError: If validation fails (unauthorized, invalid status
504                transition, date conflicts, etc.).
505        """
506        rental = Rental.query.get(rental_id)
507        if not rental:
508            raise ValueError("Rental not found")
509
510        if actor_user_id is not None and actor_user_id not in [rental.renter_id, rental.vendor_id]:
511            raise ValueError("Not authorized to modify this rental")
512
513        old_snapshot = {
514            'location': rental.location,
515            'agreed_price': float(rental.agreed_price),
516            'meeting_lat': rental.meeting_lat,
517            'meeting_lng': rental.meeting_lng,
518            'start_date': rental.start_date,
519            'end_date': rental.end_date,
520        }
521
522        if status:
523            if status not in VALID_RENTAL_STATUSES:
524                raise ValueError("Invalid rental status")
525
526            if status == 'cancelled':
527                if actor_user_id is None:
528                    raise ValueError("actor_user_id is required when canceling a rental")
529                now = datetime.utcnow()
530                if rental.status == 'requesting':
531                    if actor_user_id != rental.renter_id:
532                        raise ValueError("Only the renter can cancel a rental request")
533                elif rental.status == 'active':
534                    if actor_user_id not in [rental.renter_id, rental.vendor_id]:
535                        raise ValueError("Only rental participants can cancel this rental")
536                    if now >= rental.start_date:
537                        raise ValueError("Active rentals can only be canceled before they start")
538                else:
539                    raise ValueError("Only pending or pre-start active rentals can be canceled")
540
541            if status == 'returned':
542                if actor_user_id is None:
543                    raise ValueError("actor_user_id is required when marking a rental as returned")
544                if actor_user_id != rental.vendor_id:
545                    raise ValueError("Only the vendor can mark a rental as returned")
546                if rental.status != 'active':
547                    raise ValueError("Only active rentals can be marked as returned")
548
549                now = datetime.utcnow()
550                if now <= rental.end_date:
551                    raise ValueError("Rental can only be marked as returned after the end date has passed")
552
553            rental.status = status
554        if location:
555            rental.location = location
556        if agreed_price:
557            rental.agreed_price = agreed_price
558        if deleted is not None:
559            rental.deleted = deleted
560        if meeting_lat is not None:
561            rental.meeting_lat = meeting_lat
562        if meeting_lng is not None:
563            rental.meeting_lng = meeting_lng
564        if start_date is not None:
565            rental.start_date = RentalService._normalize_datetime(start_date, 'start_date')
566        if end_date is not None:
567            rental.end_date = RentalService._normalize_datetime(end_date, 'end_date')
568
569        if start_date is not None or end_date is not None:
570            minimum_start_date = datetime.utcnow() + RentalService.MIN_REQUEST_LEAD_TIME
571            if rental.start_date < minimum_start_date:
572                raise ValueError("Start date must be at least 2 hours in the future")
573
574        if rental.end_date <= rental.start_date:
575            raise ValueError("End date must be after start date")
576
577        details_changed = any([
578            old_snapshot['location'] != rental.location,
579            old_snapshot['agreed_price'] != float(rental.agreed_price),
580            old_snapshot['meeting_lat'] != rental.meeting_lat,
581            old_snapshot['meeting_lng'] != rental.meeting_lng,
582            old_snapshot['start_date'] != rental.start_date,
583            old_snapshot['end_date'] != rental.end_date,
584        ])
585
586        if details_changed:
587            if rental.status == 'active':
588                renegotiation_deadline = datetime.utcnow() + RentalService.MIN_RENEGOTIATION_LEAD_TIME
589                if rental.start_date <= renegotiation_deadline:
590                    raise ValueError("Active rentals can only be renegotiated if they start more than one week from now")
591
592            # Any details change requires both parties to re-approve.
593            rental.renter_approved = False
594            rental.vendor_approved = False
595            if actor_user_id == rental.renter_id:
596                rental.renter_approved = True
597            if actor_user_id == rental.vendor_id:
598                rental.vendor_approved = True
599
600            if rental.status in {'requesting', 'active'}:
601                rental.status = 'requesting'
602
603        if approve:
604            if actor_user_id is None:
605                raise ValueError("actor_user_id is required when approving")
606            if rental.status in {'requesting', 'active'}:
607                rental.renter_approved = True
608            if actor_user_id == rental.vendor_id:
609                rental.vendor_approved = True
610
611        # Enforce active status only when both parties approved.
612        if rental.status == 'active' and not (rental.renter_approved and rental.vendor_approved):
613            raise ValueError("Both renter and vendor must approve before rental can be active")
614
615        if rental.status in {'requesting', 'active'}:
616            if rental.renter_approved and rental.vendor_approved:
617                rental.status = 'active'
618            else:
619                rental.status = 'requesting'
620
621        db.session.commit()
622        return rental

642        Args:
643            equipment_name: The equipment name to search for.
644            location: A location string containing city and state.
645
646        Returns:
647            Average price as a float, or None if no matching data.
648        """
649        if not equipment_name or not location:
650            return None
651        
652        # Parse city and state from location string
653        # Location format is typically: "Street Address, City, State Zip"
654        # We'll extract city and state
655        city_state = RentalService._extract_city_state(location)
656        if not city_state:
657            return None
658        
659        city, state = city_state
660        
661        # Query for average price of rentals with this equipment in this city/state
662        result = db.session.query(func.avg(Rental.agreed_price))\
663            .join(Rental.equipment_list)\
664            .join(Equipment)\
665            .filter(Equipment.name == equipment_name)\
666            .filter(Rental.location.like(f'%{city}%'))\
667            .filter(Rental.location.like(f'%{state}%'))\
668            .scalar()
669        
670        return float(result) if result else None

672    @staticmethod
673    def _extract_city_state(location):
674        """Extract city and state from a location string.
675
676        Handles both comma-separated ("Street, City, State ZIP") and
677        newline-separated ("Street\\nCity, State ZIP") formats.
678
679        Args:
680            location: The location string to parse.
681
682        Returns:
683            Tuple of (city, state), or None if parsing fails.
684        """
685        # Handle both comma-separated and newline-separated formats
686        if '\n' in location:
687            # Multi-line format: "Street\nCity, State ZIP"
688            lines = [line.strip() for line in location.split('\n') if line.strip()]
689            if len(lines) >= 2:
690                # Last line should be "City, State ZIP"
691                city_state_line = lines[-1]
692                parts = [part.strip() for part in city_state_line.split(',')]
693                if len(parts) >= 2:
694                    city = parts[0]
695                    state_zip = parts[1].split()
696                    if state_zip:
697                        state = state_zip[0]
698                        return city, state
699        else:
700            # Comma-separated format: "Street, City, State ZIP" or "City, State"
701            parts = [part.strip() for part in location.split(',')]
702            if len(parts) >= 2:
703                if len(parts) >= 3:
704                    # "Street, City, State ZIP"
705                    city = parts[-2]
706                    state_zip = parts[-1].split()
707                else:
708                    # "City, State"
709                    city = parts[0]
710                    state_zip = parts[1].split()
711                
712                if state_zip:
713                    state = state_zip[0]
714                    return city, state
715        return None

 41    @staticmethod
 42    def _normalize_datetime(date_value, field_name):
 43        """Parse and normalize a date/datetime value to a naive UTC datetime.
 44
 45        Accepts datetime objects, date objects, and ISO 8601 strings.
 46        Timezone-aware values are converted to UTC and made naive.
 47
 48        Args:
 49            date_value: The value to normalize.
 50            field_name: Name of the field (for error messages).
 51
 52        Returns:
 53            A naive datetime in UTC, or None if date_value is None.
 54
 55        Raises:
 56            ValueError: If the value cannot be parsed.
 57        """
 58        if date_value is None:
 59            return None
 60
 61        if isinstance(date_value, datetime):
 62            value = date_value
 63        elif isinstance(date_value, date):
 64            value = datetime.combine(date_value, time.min)
 65        elif isinstance(date_value, str):
 66            normalized_value = date_value.strip().replace('Z', '+00:00')
 67            try:
 68                value = datetime.fromisoformat(normalized_value)
 69            except ValueError:
 70                try:
 71                    parsed_date = datetime.strptime(date_value, '%Y-%m-%d').date()
 72                except ValueError:
 73                    raise ValueError(f"Invalid {field_name} format. Use ISO 8601 datetime values")
 74                value = datetime.combine(parsed_date, time.min)
 75        else:
 76            raise ValueError(f"Invalid {field_name} format. Use ISO 8601 datetime values")
 77
 78        if value.tzinfo is not None:
 79            value = value.astimezone(timezone.utc).replace(tzinfo=None)
 80
 81        return value

 84    def _format_datetime(value):
 85        """Format a datetime as a human-readable UTC string.
 86
 87        Args:
 88            value: A datetime to format.
 89
 90        Returns:
 91            String like "Apr 21, 2026, 2:30 PM UTC".
 92        """
 93        if value.tzinfo is None:
 94            value = value.replace(tzinfo=timezone.utc)
 95        else:
 96            value = value.astimezone(timezone.utc)
 97        hour = value.hour % 12 or 12
 98        minute = f'{value.minute:02d}'
 99        period = 'AM' if value.hour < 12 else 'PM'
100        return f"{value.strftime('%b')} {value.day}, {value.year}, {hour}:{minute} {period} UTC"

132    def create_rental(
133        renter_id,
134        vendor_id,
135        agreed_price,

142        meeting_lat=None,
143        meeting_lng=None,
144    ):
145        """Create a new rental request with equipment.
146
147        Validates participants, dates, equipment ownership, and availability.
148        The renter is auto-approved; the vendor must separately approve.
149
150        Args:
151            renter_id: Primary key of the renter User.
152            vendor_id: Primary key of the vendor User.
153            agreed_price: Negotiated price in USD.
154            start_date: Rental start (ISO string or datetime, min 2 hours ahead).
155            end_date: Rental end (ISO string or datetime, must be after start).
156            location: Meeting location text.
157            status: Initial status (default 'requesting').
158            deleted: Soft-delete flag.
159            equipment_ids: List of Equipment primary keys to include.
160            meeting_lat: Latitude of meeting point.
161            meeting_lng: Longitude of meeting point.
162
163        Returns:
164            The created Rental instance.
165
166        Raises:
167            ValueError: If validation fails (missing fields, self-rental,
168                date conflicts, equipment not found or not owned by vendor).
169        """
170        if not all([renter_id, vendor_id, agreed_price]):
171            raise ValueError("renter_id, vendor_id, and agreed_price are required")
172        if renter_id == vendor_id:
173            raise ValueError("Renter and vendor cannot be the same user")
174
175        minimum_start_date = datetime.utcnow() + RentalService.MIN_REQUEST_LEAD_TIME
176
177        # If dates are omitted by the request flow, default to a 1-day request window.
178        if not start_date:
179            start_date = minimum_start_date
180        else:
181            start_date = RentalService._normalize_datetime(start_date, 'start_date')
182
183        if not end_date:
184            end_date = start_date + timedelta(days=1)
185        else:
186            end_date = RentalService._normalize_datetime(end_date, 'end_date')
187
188        if start_date < minimum_start_date:
189            raise ValueError("Start date must be at least 2 hours in the future")
190
191        if end_date <= start_date:
192            raise ValueError("End date must be after start date")
193        if status not in VALID_RENTAL_STATUSES:
194            raise ValueError("Invalid rental status")
195
196        requested_equipment_ids = []
197        if equipment_ids and isinstance(equipment_ids, list):
198            requested_equipment_ids.extend(equipment_ids)
199
200        # Normalize to integer IDs and flatten any accidental nested lists.
201        normalized_equipment_ids = []
202        for raw_id in requested_equipment_ids:
203            if isinstance(raw_id, list):
204                normalized_equipment_ids.extend(raw_id)
205            else:
206                normalized_equipment_ids.append(raw_id)
207
208        try:
209            normalized_equipment_ids = [int(eid) for eid in normalized_equipment_ids]
210        except (TypeError, ValueError):
211            raise ValueError("equipment_ids must be integers")
212
213        # Deduplicate while preserving order.
214        deduped_equipment_ids = list(dict.fromkeys(normalized_equipment_ids))
215        selected_equipment = []
216        for eid in deduped_equipment_ids:
217            equipment = Equipment.query.get(eid)
218            if not equipment:
219                raise ValueError(f"Equipment not found: {eid}")
220            if equipment.owner_id != vendor_id:
221                raise ValueError(f"Equipment does not belong to selected vendor: {eid}")
222            conflict = RentalService._equipment_has_conflict(equipment.id, start_date, end_date)
223            if conflict:
224                raise ValueError(f"Equipment is unavailable for selected dates: {equipment.name}")
225            selected_equipment.append(equipment)
226        
227        rental = Rental(
228            renter_id=renter_id,
229            vendor_id=vendor_id,
230            agreed_price=agreed_price,
231            start_date=start_date,
232            end_date=end_date,
233            location=location,
234            meeting_lat=meeting_lat,
235            meeting_lng=meeting_lng,
236            status=status,
237            renter_approved=True,
238            vendor_approved=False,
239            deleted=deleted,
240            renter_reviewed=False,
241            vendor_reviewed=False
242        )
243        db.session.add(rental)
244        db.session.flush()
245
246        for equipment in selected_equipment:
247            db.session.add(RentalHasEquipment(equipment_id=equipment.id, rental_id=rental.id, equipment_reviewed=False))
248
249        db.session.commit()
250        return rental

467        rental_id,
468        status=None,
469        location=None,
470        agreed_price=None,
471        deleted=None,
472        meeting_lat=None,
473        meeting_lng=None,
474        start_date=None,
475        end_date=None,
476        actor_user_id=None,
477        approve=False,
478    ):
479        """Update a rental's status, terms, and/or approval.
480
481        Handles the full spectrum of rental updates: status transitions,
482        term changes (price, dates, location), and approval management.
483        When terms change, both parties must re-approve. Enforces business
484        rules like cancellation permissions, return timing, and
485        renegotiation lead time.
486
487        Args:
488            rental_id: The rental's primary key.
489            status: New status value.
490            location: New meeting location text.
491            agreed_price: New agreed price.
492            deleted: New soft-delete flag.
493            meeting_lat: New meeting latitude.
494            meeting_lng: New meeting longitude.
495            start_date: New start date (ISO string or datetime).
496            end_date: New end date (ISO string or datetime).
497            actor_user_id: Primary key of the user performing the update.
498            approve: If True, record the actor's approval.
499
500        Returns:
501            The updated Rental instance.
502
503        Raises:
504            ValueError: If validation fails (unauthorized, invalid status
505                transition, date conflicts, etc.).
506        """
507        rental = Rental.query.get(rental_id)
508        if not rental:
509            raise ValueError("Rental not found")
510
511        if actor_user_id is not None and actor_user_id not in [rental.renter_id, rental.vendor_id]:
512            raise ValueError("Not authorized to modify this rental")
513
514        old_snapshot = {
515            'location': rental.location,
516            'agreed_price': float(rental.agreed_price),
517            'meeting_lat': rental.meeting_lat,
518            'meeting_lng': rental.meeting_lng,
519            'start_date': rental.start_date,
520            'end_date': rental.end_date,
521        }
522
523        if status:
524            if status not in VALID_RENTAL_STATUSES:
525                raise ValueError("Invalid rental status")
526
527            if status == 'cancelled':
528                if actor_user_id is None:
529                    raise ValueError("actor_user_id is required when canceling a rental")
530                now = datetime.utcnow()
531                if rental.status == 'requesting':
532                    if actor_user_id != rental.renter_id:
533                        raise ValueError("Only the renter can cancel a rental request")
534                elif rental.status == 'active':
535                    if actor_user_id not in [rental.renter_id, rental.vendor_id]:
536                        raise ValueError("Only rental participants can cancel this rental")
537                    if now >= rental.start_date:
538                        raise ValueError("Active rentals can only be canceled before they start")
539                else:
540                    raise ValueError("Only pending or pre-start active rentals can be canceled")
541
542            if status == 'returned':
543                if actor_user_id is None:
544                    raise ValueError("actor_user_id is required when marking a rental as returned")
545                if actor_user_id != rental.vendor_id:
546                    raise ValueError("Only the vendor can mark a rental as returned")
547                if rental.status != 'active':
548                    raise ValueError("Only active rentals can be marked as returned")
549
550                now = datetime.utcnow()
551                if now <= rental.end_date:
552                    raise ValueError("Rental can only be marked as returned after the end date has passed")
553
554            rental.status = status
555        if location:
556            rental.location = location
557        if agreed_price:
558            rental.agreed_price = agreed_price
559        if deleted is not None:
560            rental.deleted = deleted
561        if meeting_lat is not None:
562            rental.meeting_lat = meeting_lat
563        if meeting_lng is not None:
564            rental.meeting_lng = meeting_lng
565        if start_date is not None:
566            rental.start_date = RentalService._normalize_datetime(start_date, 'start_date')
567        if end_date is not None:
568            rental.end_date = RentalService._normalize_datetime(end_date, 'end_date')
569
570        if start_date is not None or end_date is not None:
571            minimum_start_date = datetime.utcnow() + RentalService.MIN_REQUEST_LEAD_TIME
572            if rental.start_date < minimum_start_date:
573                raise ValueError("Start date must be at least 2 hours in the future")
574
575        if rental.end_date <= rental.start_date:
576            raise ValueError("End date must be after start date")
577
578        details_changed = any([
579            old_snapshot['location'] != rental.location,
580            old_snapshot['agreed_price'] != float(rental.agreed_price),
581            old_snapshot['meeting_lat'] != rental.meeting_lat,
582            old_snapshot['meeting_lng'] != rental.meeting_lng,
583            old_snapshot['start_date'] != rental.start_date,
584            old_snapshot['end_date'] != rental.end_date,
585        ])
586
587        if details_changed:
588            if rental.status == 'active':
589                renegotiation_deadline = datetime.utcnow() + RentalService.MIN_RENEGOTIATION_LEAD_TIME
590                if rental.start_date <= renegotiation_deadline:
591                    raise ValueError("Active rentals can only be renegotiated if they start more than one week from now")
592
593            # Any details change requires both parties to re-approve.
594            rental.renter_approved = False
595            rental.vendor_approved = False
596            if actor_user_id == rental.renter_id:
597                rental.renter_approved = True
598            if actor_user_id == rental.vendor_id:
599                rental.vendor_approved = True
600
601            if rental.status in {'requesting', 'active'}:
602                rental.status = 'requesting'
603
604        if approve:
605            if actor_user_id is None:
606                raise ValueError("actor_user_id is required when approving")
